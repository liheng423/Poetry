% ==================== 核心命令 ====================
% 用法:
% \PoemWithNote{標題}{詩文內容（每行以 \par 換行）}{註釋，可留空}
\newcommand{\PoemWithNote}[3]{%
  \newpage
  \begin{center}
    \begin{minipage}{0.9\textwidth}
      \begin{flushleft}
        \section{#1}
      \end{flushleft}

      \vspace*{\fill}
      \begin{center}
        \begin{minipage}[c][0.7\textheight][c]{0.8\textwidth}
          \centering
          \LARGE #2
        \end{minipage}
      \end{center}
      \vspace*{\fill}

      \if\relax\detokenize{#3}\relax
      \else
        \footnotetext{#3}
      \fi
    \end{minipage}
  \end{center}
}

% 用法:
% \PoemFigure[旋轉角度][圖片寬度]{圖片路徑}{圖題}{label}
% 例:
% \PoemFigure[90][0.8\textwidth]{../img/anshun.png}{山寺夜景}{fig:example}
\NewDocumentCommand{\PoemFigure}{ O{0} O{0.8\textwidth} m m m }{%
  \newpage
  \vspace*{\fill}
  \begin{center}
    \begin{figure}[h!]
      \centering
      \includegraphics[angle=#1, width=#2]{#3}
      \caption{#4}
      \label{#5}
    \end{figure}
  \end{center}
  \vspace*{\fill}
}
